CREATE TABLE "ADMINISTRATOR"."FUNCTION_DOCTYPE" (
		"ID" DECIMAL(5 , 0) NOT NULL GENERATED ALWAYS AS IDENTITY ( START WITH 1 INCREMENT BY 1 MINVALUE -99999 MAXVALUE 99999 NO CYCLE CACHE 20), 
		"FUNCTION" VARCHAR(50 OCTETS) NOT NULL, 
		"DOCUMENT_TYPE" VARCHAR(500 OCTETS), 
		"FIELD_STATUS" VARCHAR(10 OCTETS)
	);
	



CREATE TABLE "ADMINISTRATOR"."UNIQUE_CODE" (
		"ID" DECIMAL(5 , 0) NOT NULL GENERATED ALWAYS AS IDENTITY ( START WITH 1 INCREMENT BY 1 MINVALUE -99999 MAXVALUE 99999 NO CYCLE CACHE 20 ), 
		"UNIQUE_CODE" VARCHAR(50 OCTETS) NOT NULL, 
		"ENTITY" VARCHAR(500 OCTETS), 
		"REGION" VARCHAR(50 OCTETS), 
		"STATE" VARCHAR(50 OCTETS), 
		"CITY" VARCHAR(50 OCTETS), 
		"PRIMARY_USER" VARCHAR(100 OCTETS), 
		 PRIMARY KEY
		("UNIQUE_CODE")
	);
